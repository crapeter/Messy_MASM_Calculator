
.386
.model flat,stdcall
.stack 4096
ExitProcess proto, dwExitCode:dword

include irvine32.inc

.data
	myarr sbyte -5, -6, -7, -4

.code
main proc
	mov edi, offset myarr
	mov ecx, SIZEOF myarr
	mov dl, 0
L1:
	mov al, [edi]
    neg al
	add dl, al
	add edi, type myarr
	loop L1

	mov eax, dl
	call WriteInt

	invoke ExitProcess, 0
main endp
end main
