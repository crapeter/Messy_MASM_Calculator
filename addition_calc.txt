
.386
.model flat,stdcall
.stack 4096
ExitProcess proto, dwExitCode:dword

include irvine32.inc

.data
    decVal1 dword 1
    decVal2 dword 1
	sum dword ?

.code
main proc
	call readint
	mov decVal1, eax
	call readint
	mov decVal2, eax

	mov ebx, 0
	add ebx, decVal1
	add ebx, decVal2
	mov sum, ebx
	mov eax, sum
	call writeint
	

	invoke ExitProcess, 0
main endp
end main





;displaying how to use this
	mov edx, offset usage
	call WriteString
	call crlf

;getting the operation type
	call readchar
	mov operation, al

;getting the numbers
	call readdec
	mov decVal1, eax
	call readdec
	mov decVal2, eax

	cmp al, 'a'
	jmp Addition
	cmp al, 's'
	jmp Subtraction
	cmp al, 'm'
	jmp Multiplication
	cmp al, 'd'
	jmp Division

Addition:
	mov ebx, decVal1
	add ebx, decVal2
	mov result, ebx
	mov eax, result
	call writedec
	jmp End
Subtraction:
	mov ebx, decVal1
	sub ebx, decVal2
	mov result, ebx
	mov eax, result
	call writedec
	jmp End
Multiplication:
	mov eax, decVal1
	mul decVal2
	call writedec
	jmp End
Division:
	cmp decVal2, 0
	jmp DivideByZero
	mov eax, decVal1
	xor edx, edx
	div decVal2
	call writedec
	jmp End

DivideByZero:
	mov edx, offset divError
	call WriteString
	call crlf
    jmp End

End:
