
.386
.model flat,stdcall
.stack 4096
ExitProcess proto, dwExitCode:dword

include irvine32.inc

.data
	firstNum byte "Enter the first number: ", 0
	secondNum byte "Enter the second number: ", 0
	addition_title byte "Addition result: ", 0 
	subtraction_title byte "Subtraction result: ", 0
    num1 dword ?
    num2 dword ?

.code
main proc
	;getting the first number
	mov edx, offset firstNum
	call writestring
	call readint
	mov num1, eax

	;getting the second number
	mov edx, offset secondNum
	call writestring
	call readint
	mov num2, eax
	call crlf

	call addition
	call subtraction


	invoke ExitProcess, 0
main endp

addition proc
	mov eax, num1
	add eax, num2
	mov edx, offset addition_title
	call writestring
	call writeint
	call crlf
	ret
addition endp

subtraction proc
	mov eax, num1
	sub eax, num2
	mov edx, offset subtraction_title
	call writestring
	call writeint
	call crlf
	ret
subtraction endp

end main
