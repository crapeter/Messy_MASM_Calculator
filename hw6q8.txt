
.386
;.model flat,stdcall
.stack 4096
ExitProcess proto, dwExitCode:dword

include irvine32.inc

.data
	exam_arr byte 70, 80, 90, 85, 30, 40, 95, 60, 85, 78
	examsList byte "List of exam grades:", 0
	highestScore byte "Highest exam score is: ", 0
	lowestScore byte "Lowest exam score is: ", 0
	space byte " ", 0
	temp byte ?
	highest dword 0
	lowest dword 100

.code
main proc
	mov edx, offset examsList
	call writestring
	call crlf
	mov ecx, lengthof exam_arr
	mov ebx, 0
	L1:
        mov al, [exam_arr + ebx]
        mov temp, al
        movzx eax, temp
		mov edx, offset space
		call writedec
		call writestring
        cmp eax, highest
        jg highNum
        cmp eax, lowest
        jl lowNum
        jmp continueLoop
		highNum:
			mov highest, eax
			jmp continueLoop
		lowNum:
			mov lowest, eax
		continueLoop:
			add ebx, type exam_arr
			loop L1

	call crlf
	call crlf
	mov edx, offset highestScore
	mov eax, highest
	call writestring
	call writedec
	call crlf
	mov edx, offset lowestScore
	mov eax, lowest
	call writestring
	call writedec
	call crlf
	invoke ExitProcess, 0
main endp
end main
